  @pytest.fixture
  def ckp_factory(mock_neo4j: InMemoryNeo4jTracker):
      """Create checkpoints directly in mock_neo4j (no HTTP).

      Returns a callable make_ckp(experiment_id, epoch, metrics) -> checkpoint_id.
      Bypasses HTTP to avoid double-patching conflicts when used alongside lineage_client.

      Usage:
          ckp_id = ckp_factory(experiment_id=ctx.experiment_id, epoch=1, metrics={"loss": 0.5})
      """
      import uuid as _uuid
      from graph_lineage.data_classes.neo4j.nodes.checkpoint import Checkpoint

      def make_ckp(
          experiment_id: str,
          epoch: int = 1,
          metrics: dict | None = None,
      ) -> str:
          ckp_id = str(_uuid.uuid4())
          ckp = Checkpoint(
              id=ckp_id,
              name=f"checkpoint-epoch-{epoch}",
              epoch=epoch,
              run=1,
              uri=f"/output/checkpoint-epoch-{epoch}/{ckp_id[:8]}",
              metrics=metrics or {"loss": 0.5},
              derived_from="",
              is_merging=False,
          )
          mock_neo4j.create_checkpoint_node(ckp)
          mock_neo4j.create_checkpoint_edge(experiment_id, ckp_id)
          return ckp_id

      return make_ckp


  # ─────────────────────────────────────────────────────────────────────────────
  # Codebase Mutator Fixture
  # ─────────────────────────────────────────────────────────────────────────────


  @pytest.fixture
  def codebase_mutator(test_project: Path):
      """Mutate test_project files to trigger BRANCH detection on next pre_execution().

      The LineageClient computes a snapshot hash of config.yml, train.py, prepare.py,
      requirements.txt. Changing config.yml produces a different hash → server detects BRANCH.

      Usage:
          codebase_mutator.change_config({"learning_rate": 2e-4})
          ctx_branch = lineage_client.pre_execution()  # will be BRANCH
      """

      class CodebaseMutator:
          def __init__(self, project_root: Path) -> None:
              self._root = project_root

          def change_config(self, updates: dict) -> None:
              """Merge updates into config.yml training section."""
              config_path = self._root / "config.yml"
              config = yaml.safe_load(config_path.read_text()) or {}
              config.setdefault("model", {}).setdefault("training", {}).update(updates)
              config_path.write_text(yaml.dump(config))

          def reset_config(self) -> None:
              """Restore config.yml to its original content (re-create from scratch)."""
              config_path = self._root / "config.yml"
              config_path.write_text(
                  yaml.dump({
                      "experiment": {"name": "test-train", "uri": str(self._root)},
                      "model": {
                          "model_uri": "/nfs/llama-7b",
                          "model_id": "llama-7b-base",
                          "training": {"learning_rate": 1e-4, "per_device_train_batch_size": 1},
                      },
                      "output": {"output_dir": str(self._root / "output")},
                  })
              )

      return CodebaseMutator(test_project)
  ```

  """Sequential integration scenarios S-01 through S-02.

  Tests multi-step lineage chains using real server logic (via TestHttpConnector)
  with InMemoryNeo4jTracker for graph assertions.
  """

  from __future__ import annotations

  import sys
  from pathlib import Path

  import pytest

  sys.path.insert(0, str(Path(__file__).resolve().parent.parent / "graph_lineage" / "setups" / "_base"))

  from modules.lineage.client import LineageClient
  from tests.mock_neo4j import InMemoryNeo4jTracker


  class TestS01NewRetryRetry:
      """S-01: NEW → RETRY → RETRY persistent failure chain.

      Verifies a linear RETRY_OF chain with no checkpoints.
      Also covers Issue #3: description must be preserved from experiment.yml in subsequent calls.
      Graph: e3 -[RETRY_OF]-> e2 -[RETRY_OF]-> e1
      """

      def test_s01_new_retry_retry(
          self,
          lineage_client: LineageClient,
          mock_neo4j: InMemoryNeo4jTracker,
      ):
          # Step 1 — NEW
          ctx1 = lineage_client.pre_execution()
          assert ctx1.strategy == "NEW"
          lineage_client.post_execution(ctx1, status="FAILED")

          # Step 2 — RETRY (same codebase, no checkpoint)
          ctx2 = lineage_client.pre_execution()
          assert ctx2.strategy == "RETRY", (
              f"Expected RETRY after failed NEW, got {ctx2.strategy!r}"
          )
          # Issue #3: description must be preserved — client reads it from experiment.yml
          # (client.py line 173: description=exp_data.get("description"))
          assert ctx2.description is not None, (
              "Issue #3: description must not be None in RETRY chain — "
              "client must read description from experiment.yml written after NEW pre_execution"
          )
          lineage_client.post_execution(ctx2, status="FAILED")

          # Step 3 — another RETRY
          ctx3 = lineage_client.pre_execution()
          assert ctx3.strategy == "RETRY", (
              f"Expected RETRY after second failure, got {ctx3.strategy!r}"
          )
          lineage_client.post_execution(ctx3, status="FAILED")

          # Graph assertions
          mock_neo4j.assert_experiment_count(3)
          mock_neo4j.assert_edges_count_of_type("RETRY_OF", 2)
          mock_neo4j.assert_edge_exists(ctx2.experiment_id, ctx1.experiment_id, "RETRY_OF")
          mock_neo4j.assert_edge_exists(ctx3.experiment_id, ctx2.experiment_id, "RETRY_OF")

          # No checkpoints were produced
          mock_neo4j.assert_checkpoint_count(0)

          # No DERIVED_FROM edges
          mock_neo4j.assert_edges_count_of_type("DERIVED_FROM", 0)

          # Anti-circular: no self-loops or cycles
          all_retry_edges = mock_neo4j.get_edges_of_type("RETRY_OF")
          for frm, to, _, _ in all_retry_edges:
              assert frm != to, f"Self-loop detected on RETRY_OF: {frm}"


  class TestS02NewBranchBranch:
      """S-02: NEW → BRANCH → BRANCH (config changes at each step).

      Verifies a two-level derivation tree where each DERIVED_FROM edge carries diff_patch.
      Graph: e1 <-[DERIVED_FROM]- e2 <-[DERIVED_FROM]- e3
      """

      def test_s02_new_branch_branch(
          self,
          lineage_client: LineageClient,
          mock_neo4j: InMemoryNeo4jTracker,
          codebase_mutator,
      ):
          # Step 1 — NEW baseline
          ctx1 = lineage_client.pre_execution()
          assert ctx1.strategy == "NEW"
          lineage_client.post_execution(ctx1, status="COMPLETED")

          # Step 2 — mutate config → BRANCH
          codebase_mutator.change_config({"learning_rate": 2e-4})
          ctx2 = lineage_client.pre_execution()
          assert ctx2.strategy == "BRANCH", (
              f"Expected BRANCH after config change, got {ctx2.strategy!r}"
          )
          assert len(ctx2.changed_files) > 0, "BRANCH must report changed files"
          lineage_client.post_execution(ctx2, status="COMPLETED")

          # Step 3 — another config mutation → second BRANCH
          codebase_mutator.change_config({"num_epochs": 5})
          ctx3 = lineage_client.pre_execution()
          assert ctx3.strategy == "BRANCH", (
              f"Expected BRANCH after second config change, got {ctx3.strategy!r}"
          )

          # Graph assertions: 3 experiments, 2 DERIVED_FROM edges
          mock_neo4j.assert_experiment_count(3)
          mock_neo4j.assert_edges_count_of_type("DERIVED_FROM", 2)
          mock_neo4j.assert_edge_exists(ctx2.experiment_id, ctx1.experiment_id, "DERIVED_FROM")
          mock_neo4j.assert_edge_exists(ctx3.experiment_id, ctx2.experiment_id, "DERIVED_FROM")

          # Both DERIVED_FROM edges must carry a non-empty diff_patch
          branch_edges = mock_neo4j.get_edges_of_type("DERIVED_FROM")
          for frm, to, rel_type, props in branch_edges:
              assert props.get("diff_patch"), (
                  f"DERIVED_FROM edge {frm}->{to} missing diff_patch"
              )

          # No RETRY edges
          mock_neo4j.assert_edges_count_of_type("RETRY_OF", 0)
  ```

# To trigger RESUME strategy in lineage_client:
#   1. Call ckp_factory(experiment_id, epoch) to create a checkpoint in mock_neo4j
#   2. Update test_project/.lineage/experiment.yml to set checkpoint_resume_from = ckp_id

  ```python
  def merge_checkpoints(
      self,
      sources: list[str],
      experiment_id: str | None = None,
  ) -> str:
      """Create a merged Checkpoint node with MERGED_FROM edges to each source.

      Edge direction: merged_ckp -[MERGED_FROM]-> source_ckp (per TEST_EXTENSION_PLAN.md).

      Args:
          sources: List of source checkpoint IDs
          experiment_id: Optional — if provided, creates PRODUCED edge from experiment

      Returns:
          The merged checkpoint ID
      """
      from uuid import uuid4 as _uuid4
      from graph_lineage.data_classes.neo4j.nodes.checkpoint import Checkpoint

      merged_id = str(_uuid4())
      merged = Checkpoint(
          id=merged_id,
          name="merged-checkpoint",
          epoch=0,
          run=1,
          uri=f"/output/merged/{merged_id[:8]}",
          metrics={},
          derived_from="",
          is_merging=False,
      )
      self.create_checkpoint_node(merged)
      for source_id in sources:
          self.create_edge(merged_id, source_id, "MERGED_FROM")
      return merged_id

  def promote_checkpoint(
      self,
      ckp_id: str,
      model_name: str,
      model_type: str = "adapter",
  ) -> str:
      """Create a Model record and PROMOTED_TO edge from checkpoint.

      Edge direction: checkpoint -[PROMOTED_TO]-> model.

      Args:
          ckp_id: Checkpoint to promote
          model_name: Name of the resulting model
          model_type: Type of model (default "adapter")

      Returns:
          The model ID
      """
      from uuid import uuid4 as _uuid4

      model_id = str(_uuid4())
      self.models[model_id] = {
          "id": model_id,
          "model_name": model_name,
          "type": model_type,
      }
      self.create_edge(ckp_id, model_id, "PROMOTED_TO")
      return model_id

  def get_model(self, model_id: str) -> dict[str, Any] | None:
      """Get raw model data by ID."""
      return self.models.get(model_id)

  def get_model_by_name(self, model_name: str) -> dict[str, Any] | None:
      """Find a model by name."""
      for m in self.models.values():
          if m.get("model_name") == model_name:
              return m
      return None

  def assert_model_exists(
      self,
      model_name: str,
      model_type: str | None = None,
  ) -> None:
      """Assert a model with given name exists.

      Raises:
          AssertionError if model not found or type mismatch
      """
      m = self.get_model_by_name(model_name)
      assert m is not None, f"Model '{model_name}' not found. Available: {list(self.models.values())}"
      if model_type is not None:
          assert m.get("type") == model_type, (
              f"Model '{model_name}' type mismatch: expected {model_type!r}, got {m.get('type')!r}"
          )
  ```

  ```python
  @pytest.fixture
  def merge_client(mock_neo4j: InMemoryNeo4jTracker):
      """Thin wrapper around mock_neo4j merge/promote for scenario tests.

      Usage:
          merged_ckp_id = merge_client.merge(sources=[ckp_a, ckp_b])
          model_id = merge_client.promote(merged_ckp_id, model_name="my-adapter")
      """

      class MergeResult:
          def __init__(self, checkpoint_id: str, is_merging: bool = False, is_usable: bool = True):
              self.checkpoint_id = checkpoint_id
              self.is_merging = is_merging
              self.is_usable = is_usable

      class _MergeClient:
          def __init__(self, tracker: InMemoryNeo4jTracker) -> None:
              self._tracker = tracker

          def merge(
              self,
              sources: list[str],
              experiment_id: str | None = None,
          ) -> MergeResult:
              """Merge source checkpoints into a new checkpoint node."""
              merged_id = self._tracker.merge_checkpoints(sources, experiment_id=experiment_id)
              return MergeResult(checkpoint_id=merged_id, is_merging=False, is_usable=True)

          def promote(self, ckp_id: str, model_name: str, model_type: str = "adapter") -> str:
              """Promote a checkpoint to a Model node."""
              return self._tracker.promote_checkpoint(ckp_id, model_name, model_type=model_type)

      return _MergeClient(mock_neo4j)
  ```

  ```python
          test_project,
  ```

  ```python
      def test_s05_intra_experiment_merge(
          self,
          lineage_client: LineageClient,
  ```

  ```python
          codebase_mutator,
          test_project,

          ctx3 = lineage_client.pre_execution()
          assert ctx3.strategy == "RETRY", (
              f"Expected RETRY after BRANCH failure with no new changes, got {ctx3.strategy!r}"
          )
          ckp2 = ckp_factory(ctx3.experiment_id, epoch=2, metrics={"loss": 0.29})
          lineage_client.post_execution(ctx3, status="COMPLETED")

          # === e4: RESUME from ckp2 ===

          exp_data = yaml.safe_load(exp_yml.read_text())
          exp_data["experiment"]["checkpoint_resume_from"] = ckp2
          exp_yml.write_text(yaml.dump(exp_data))

          ctx4 = lineage_client.pre_execution()
          assert ctx4.strategy == "RESUME", (
              f"Expected RESUME (checkpoint_resume_from={ckp2!r}), got {ctx4.strategy!r}"
          )

          # === Full graph assertions ===
          mock_neo4j.assert_experiment_count(4)

          mock_neo4j.assert_edge_exists(ctx2.experiment_id, ctx1.experiment_id, "DERIVED_FROM")
          mock_neo4j.assert_edge_exists(ctx3.experiment_id, ctx2.experiment_id, "RETRY_OF")
          started_from_edges = mock_neo4j.get_edges_of_type("STARTED_FROM")
          assert any(
              frm == ctx4.experiment_id and to == ckp2
              for frm, to, _, _ in started_from_edges

              visited: set[str] = set()
              frontier = {exp_id}
              while frontier:
                  current = frontier.pop()
                  if current in visited:
                      continue
                  visited.add(current)

                      if frm == current and to not in visited:

              assert exp_id not in (visited - {exp_id}), (
                  f"Circular dependency detected: {exp_id} is its own ancestor"
              )
  ```
